#include <stdio.h>

int main() {
    int n;
    printf("양의 정수 n을 입력하세요: ");
    scanf("%d", &n); // 사용자로부터 n 값을 입력받습니다.

    if (n < 0) {
        printf("음수가 아닌 양의 정수를 입력하세요.\n");
        return 1; // 오류 코드로 프로그램 종료
    }

    int result = 1; // 결과를 저장할 변수를 1로 초기화합니다.

    for (int i = 1; i <= n; i++) {
        result *= i; // 현재 i 값을 결과에 곱합니다.
    }

    printf("%d! = %d\n", n, result); // 결과를 출력합니다.

    return 0;
}
